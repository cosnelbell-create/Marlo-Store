    let cart = [];
    let currentFilter = 'all';

    // Fonction WhatsApp avec ton nouveau lien exact
    function openWhatsApp() {
      window.open('https://wa.me/message/U3AUHWUIXGFQC1', '_blank');
    }

    function renderProducts() {
      const grid = document.getElementById('products-grid');
      const filtered = currentFilter === 'all' ? products : products.filter(p => p.category === currentFilter);
      grid.innerHTML = filtered.map(p => `
        <div class="product-card bg-white rounded-3xl overflow-hidden shadow-xl border border-gray-100">
          <img src="${p.img}" class="w-full h-72 object-cover">
          <div class="p-6">
            <span class="text-xs uppercase tracking-widest text-[var(--gold)]">${p.category}</span>
            <h3 class="text-2xl heading mt-2 mb-1">${p.name}</h3>
            <p class="text-3xl font-bold text-amber-800">${p.price.toFixed(2)} €</p>
            <p class="text-sm text-gray-600 mt-3 line-clamp-2">${p.desc}</p>
            <div class="flex gap-3 mt-6">
              <button onclick="addToCart(${p.id}); event.stopImmediatePropagation()" class="flex-1 bg-black text-white py-4 rounded-2xl hover:bg-gray-800">Ajouter au panier</button>
              <button onclick="showProductModal(${p.id}); event.stopImmediatePropagation()" class="flex-1 border border-black py-4 rounded-2xl hover:bg-gray-100">Détails</button>
            </div>
            <button onclick="openWhatsApp(); event.stopImmediatePropagation()"

              <i class="fa-brands fa-whatsapp"></i> Commander via WhatsApp
            </button>
          </div>
        </div>
      `).join('');
    }

    function showProductModal(id) {
      const p = products.find(x => x.id === id);
      document.getElementById('modal-content').innerHTML = `
        <button onclick="closeModal()" class="float-right text-4xl">×</button>
        <img src="${p.img}" class="w-full h-96 object-cover rounded-2xl mb-8">
        <h2 class="text-4xl heading">${p.name}</h2>
        <p class="text-4xl font-bold text-amber-800 mt-2">${p.price.toFixed(2)} €</p>
        <p class="mt-6 leading-relaxed">${p.desc}</p>
        <button onclick="openWhatsApp(); closeModal()" class="mt-10 w-full bg-green-500 text-white py-6 rounded-3xl text-xl flex items-center justify-center gap-4">
          <i class="fa-brands fa-whatsapp text-3xl"></i> Commander maintenant sur WhatsApp
        </button>
      `;
      document.getElementById('product-modal').classList.remove('hidden');
    }

    function closeModal() {
      document.getElementById('product-modal').classList.add('hidden');
    }

    function createFilters() {
      const cats = ['all', ...new Set(products.map(p => p.category))];
      const container = document.getElementById('category-filters');
      container.innerHTML = cats.map(cat => `
        <button onclick="filterCategory('${cat}')" 
                class="px-8 py-4 rounded-3xl font-medium ${currentFilter === cat ? 'bg-[var(--gold)] text-black' : 'bg-white border hover:border-[var(--gold)]'}">
          ${cat === 'all' ? 'Tout' : cat}
        </button>
      `).join('');
    }

    function filterCategory(cat) {
      currentFilter = cat;
      createFilters();
      renderProducts();
    }

    function addToCart(id) {
      const product = products.find(p => p.id === id);
      const existing = cart.find(item => item.id === id);
      if (existing) existing.quantity++;
      else cart.push({...product, quantity: 1});
      updateCartCount();
      alert(`${product.name} ajouté au panier !`);
    }

    function updateCartCount() {
      const count = cart.reduce((sum, item) => sum + item.quantity, 0);
      document.getElementById('cart-count').textContent = count;
    }

    function toggleCart() {
      const modal = document.getElementById('cart-modal');
      const itemsContainer = document.getElementById('cart-items');
      const totalEl = document.getElementById('cart-total');

      if (modal.classList.contains('hidden')) {
        modal.classList.remove('hidden');
        itemsContainer.innerHTML = cart.map(item => `
          <div class="flex gap-4 mb-6">
            <img src="${item.img}" class="w-20 h-20 object-cover rounded-2xl">
            <div class="flex-1">
              <p class="font-medium">${item.name}</p>
              <p class="text-sm text-gray-500">${item.price.toFixed(2)} € × ${item.quantity}</p>
              <div class="flex gap-3 mt-3">
                <button onclick="changeQuantity(${item.id}, -1)" class="text-red-600">-</button>
                <span class="font-bold">${item.quantity}</span>
                <button onclick="changeQuantity(${item.id}, 1)" class="text-green-600">+</button>
              </div>
            </div>
            <div class="font-bold">${(item.price * item.quantity).toFixed(2)} €</div>
          </div>
        `).join('') || '<p class="text-center text-gray-400 py-12">Ton panier est vide</p>';

        const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
        totalEl.textContent = total.toFixed(2) + ' €';
      } else {
        modal.classList.add('hidden');
      }
    }

    function changeQuantity(id, change) {
      const item = cart.find(i => i.id === id);
      if (!item) return;
      item.quantity += change;
      if (item.quantity < 1) cart = cart.filter(i => i.id !== id);
      toggleCart();
      updateCartCount();
    }

    function checkout() {
      if (cart.length === 0) return;
      openWhatsApp(); // ouvre directement ton nouveau lien WhatsApp
      cart = [];
      updateCartCount();
      toggleCart();
    }

    // Initialisation
    window.onload = () => {
      createFilters();
      renderProducts();
      updateCartCount();
    };
